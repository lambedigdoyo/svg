Get blob from Image Link/url

Javascript: "XMLHttpRequest()" Method

Paste Image Link here:

URL:

blobURL:

<script>
var imageLink = document.getElementById('imageLink');

function getBlobURL() {
  var url = imageLink.value;
  var xhr = new XMLHttpRequest();
  xhr.open('get', url, true);
  xhr.responseType = 'blob';
  xhr.onload = () => {
    if (xhr.status === 200) {
      var file = xhr.response;
      var urlBlob = URL.createObjectURL(file);
      document.querySelector('#blobURL').value = urlBlob;
      document.querySelector('#blobPreview').src = urlBlob;
    }
  }
  xhr.send();
  document.getElementById('imageURL').value = url;
}
imageLink.oninput = getBlobURL;
getBlobURL()
</script>
